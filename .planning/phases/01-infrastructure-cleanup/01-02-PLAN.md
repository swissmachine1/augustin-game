    The full file:
    ```javascript
    import * as Phaser from 'phaser'

    export class LevelSelectHub extends Phaser.Scene {
      constructor() {
        super('LevelSelectHub')
      }

      create() {
        const { width, height } = this.cameras.main
        const name = this.registry.get('playerName') || 'friend'

        this.add.text(width / 2, height / 2 - 60, 'LEVEL SELECT HUB', {
          fontFamily: 'monospace',
          fontSize: '48px',
          color: '#00ff88',
          fontStyle: 'bold',
        }).setOrigin(0.5)

        this.add.text(width / 2, height / 2 + 10, `Welcome, ${name}`, {
          fontFamily: 'monospace',
          fontSize: '24px',
          color: '#ffffff',
        }).setOrigin(0.5)

        this.add.text(width / 2, height / 2 + 70, '5 levels coming in Phase 2', {
          fontFamily: 'monospace',
          fontSize: '16px',
          color: '#8888aa',
        }).setOrigin(0.5)

        this.events.once('shutdown', () => {
          // No cleanup needed for placeholder hub
        }, this)
      }
    }
    ```
